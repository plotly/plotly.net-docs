Basic 3D Cone¶

In [2]:
open System
open Plotly.NET

let cone3d =
    Trace3d.initCone
        (fun cone3d ->
            cone3d?x <- [1]
            cone3d?y <- [1]
            cone3d?z <- [1]
            cone3d?u <- [1]
            cone3d?v <- [1]
            cone3d?w <- [0]
            cone3d?colorscale<-"Blues"
            cone3d?sizemode<-"absolute"
            cone3d?sizeref<- 20,
            cone3d?anchor<-"tip"
            cone3d
            )
    |> GenericChart.ofTraceObject

Multiple 3D Cones¶

In [4]:
open System
open Plotly.NET


let cone3d =
    Trace3d.initCone
        (fun cone3d ->
            cone3d?x<-[1; 2; 3]
            cone3d?y<-[1; 2; 3]
            cone3d?z<-[1; 2; 3]
            cone3d?u<-[1; 0; 0]
            cone3d?v<-[0; 3; 0]
            cone3d?w<-[0; 0; 2]
            cone3d?sizemode<-"absolute"
            cone3d?sizeref<-2
            cone3d?anchor<-"tip"
            cone3d
            )
    |> GenericChart.ofTraceObject
